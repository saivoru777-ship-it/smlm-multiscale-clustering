\section*{Supplementary Figures}

\begin{figure}[ht]
\centering
\includegraphics[width=0.8\textwidth]{figures/figS1_blinking.pdf}
\caption{\textbf{Figure S1: Blinking correction performance.}
  Reduction factors (raw localizations / merged molecules) as a function
  of blinking rate (10 replicates). Even at 20 blinks per molecule, the
  correction reduces the localization count by only $\sim$10\%, indicating
  conservative merging. Detection power (TPR) remains 1.0 across all
  blinking rates.}
\label{fig:s1}
\end{figure}

\begin{figure}[ht]
\centering
\includegraphics[width=0.8\textwidth]{figures/figS2_conditions.pdf}
\caption{\textbf{Figure S2: Realistic SMLM conditions} (10 replicates).
  \textbf{(A)}~Detection power vs.\ localization precision
  ($\sigma_{xy}$ = 5--100~nm). TPR = 1.0 across all precisions.
  \textbf{(B)}~Detection power vs.\ number of imaging frames
  (20--2000). TPR = 1.0 regardless of frame count.}
\label{fig:s2}
\end{figure}

\begin{figure}[ht]
\centering
\includegraphics[width=0.8\textwidth]{figures/figS3_condition_number.pdf}
\caption{\textbf{Figure S3: Covariance matrix condition numbers} (20
  replicates). Condition numbers of the regularized covariance matrix
  $\boldsymbol{\Sigma}_\mathrm{reg}$ as a function of the shrinkage
  parameter $\lambda$. Mean condition numbers range from 413
  ($\lambda = 0.5$) to 619 ($\lambda = 0.05$), ensuring stable matrix
  inversion across all settings.}
\label{fig:s3}
\end{figure}

\begin{figure}[ht]
\centering
\includegraphics[width=0.8\textwidth]{figures/figS4_runtime.pdf}
\caption{\textbf{Figure S4: Runtime scaling.}
  Wall-clock time for the full pipeline as a function of the number of
  molecules (500--20,000). Scaling is near-linear up to 5000 molecules
  ($\sim$51~s), with superlinear growth at larger sizes due to
  covariance estimation and mock generation costs.}
\label{fig:s4}
\end{figure}
